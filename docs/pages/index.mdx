import Head from 'next/head'
import { useEffect } from 'react'
import { useRouter } from 'next/router'

export function Redirect() {
  const router = useRouter()
  useEffect(() => {
    router.replace('/client')
  }, [])
  return (
    <Head>
      <meta httpEquiv="refresh" content="0;url=/client" />
    </Head>
  )
}

<Redirect />
